Adicionar números ao array

Array de números:

Código usado:


    var numeros = []; // Array para armazenar os números

    function adicionarNumero() {
      var input = document.getElementById("numeroInput");
      var numero = parseInt(input.value); // Converter o valor do input para um número inteiro
      
      if (!isNaN(numero)) { // Verificar se o número é válido
        numeros.push(numero); // Adicionar o número ao array
        input.value = ""; // Limpar o input
        
        exibirArrayNumeros(); // Atualizar a exibição do array
      }
    }
    
    function exibirArrayNumeros() {
      var numerosSpan = document.getElementById("numerosArray");
      numerosSpan.textContent = numeros.join(", "); // Exibir o array separado por vírgulas
    }
    
    function encontrarMaiorNumero() {
      var maior = Math.max(...numeros); // Encontrar o maior número no array
      alert("O maior número é: " + maior); // Exibir o maior número em um alerta
    }
